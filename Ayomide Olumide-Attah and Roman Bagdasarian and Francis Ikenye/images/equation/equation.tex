\[
\begin{gathered}
[U_A]_{nm}
  = \frac{1}{N}
    \sum_{k=0}^{N-1}
    \mathrm{e}^{\frac{2\pi i k (n - A m)}{N}}
    \;\equiv\;
    [F_N^{-1}G_N]_{nm} \\[1em]
[F_N]_{nm}
  = \frac{1}{\sqrt{N}}
    \mathrm{e}^{-\frac{2\pi i n m}{N}} \\[1em]
[G_N]_{nm}
  = \frac{1}{\sqrt{N}}
    \mathrm{e}^{-\frac{2\pi i A n m}{N}} \\[1em]
\end{gathered}
\]
